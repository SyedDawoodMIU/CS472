4th Corrected


        function makeArmy() {
          let shooters = [];
          let i = 0;
          while (i < 2) {
            let shooter = (function (j) {
              return function () {
                alert(j);
              };
            })(i);
            shooters.push(shooter);
            i++;
          }
          return shooters;
        }
        let army = makeArmy();
        army[0]();
