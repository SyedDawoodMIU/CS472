max(num1, num2)

Returns the largest of two numbers passed as arguments.


              "use strict";
              function max(num1, num2) {
                if (num1 > num2) {
                  return num1;
                } else {
                  return num2;
                }
              }

maxOfThree(num1, num2, num3)

Returns the largest of three numbers passed as arguments.


              "use strict";
              function maxOfThree(num1, num2, num3) {
                if (num1 > num2 && num1 > num3) {
                  return num1;
                } else if (num2 > num1 && num2 > num3) {
                  return num2;
                } else {
                  return num3;
                }
              }

isVowel(char)

Returns true if the character passed as an argument is a vowel (a, e, i, o, u).


              "use strict";
              function isVowel(char) {
                var vowels = ["a", "e", "i", "o", "u"];
                return vowels.indexOf(char.toLowerCase()) !== -1;
              }

isVowelIncluded(char)

              "use strict";
              function isVowelIncluded(char) {

reverse(str)

Returns the reversed version of a string.


                "use strict";
                function reverse(str) {
                    return str.split("").reverse().join("");
                  }

reverseUsingLoop(str)

Returns the reversed version of a string using loop.


                "use strict";
                function reverse(str) {
                    let reversed = '';
                    for (let i = str.length - 1; i >= 0; i--) {
                      reversed += str[i];
                    }
                    return reversed;
                  }

filterLongWords(words, i)

Returns an array of words longer than a given length.


                "use strict";
                function filterLongWords(words, i) {

computeSumOfSquares(numbers)

Returns the sum of the squares of all numbers in an array.


                "use strict";
                function computeSumOfSquares(numbers) {
                    return numbers.reduce(function (sum, number) {
                      return sum + number * number;
                    }, 0);
                  }

findSecondBiggestForOf(arr)

Returns the second largest number in an array using a for...of loop.


                "use strict";
                function findSecondBiggestForOf(arr) {
                    let max = arr[0];
                    let secondMax = arr[0];
                  
                    for (let num of arr) {
                      if (num > max) {
                        secondMax = max;
                        max = num;
                      } else if (num > secondMax && num != max) {
                        secondMax = num;
                      }
                    }
                  
                    return secondMax;
                  }

findSecondBiggest(arr)

Returns the second biggest number in an array.


              "use strict";
              function findSecondBiggest(arr) {
                let max = arr[0];
                let secondMax = arr[0];
                for (let i = 1; i < arr.length; i++) {
                    if (arr[i] > max) {
                      secondMax = max;
                      max = arr[i];
                    } else if (arr[i] > secondMax && arr[i] != max) {
                      secondMax = arr[i];
                    }
                  }
                
                  return secondMax;
                }

findSecondBiggestUsingReduce(arr)

Returns the second biggest number in an array using reduce.


              "use strict";
              function findSecondBiggestUsingReduce(arr) {
                const [biggest, secondBiggest] = arr.reduce(([max, secondMax], current) => {
                  if (current > max) {
                    return [current, max];
                  } else if (current > secondMax && current !== max) {
                    return [max, current];
                  }
                  return [max, secondMax];
                }, [-Infinity, -Infinity]);
                return secondBiggest;
              }

printFibo(n, a, b)

Takes n length and two starting numbers a and b and logs the Fibonacci sequence.


              "use strict";
              function printFibo(n, a, b) {
                let fib = [a, b];
                for (let i = 2; i < n; i++) {
                  fib[i] = fib[i - 1] + fib[i - 2];
                }
                console.log(fib.join(", "));
              }

Returns a new array with all strings that are at least 5 characters long and contain the letter 'a'.
