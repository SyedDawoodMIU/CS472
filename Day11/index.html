          Array.prototype.even = function () {
            return this.filter((num) => num % 2 === 0);
          };
          
          Array.prototype.odd = function () {
            return this.filter((num) => num % 2 !== 0);
          };
          
          const numArray = [1, 2, 3, 4, 5, 6, 7, 8];
          console.log(numArray.even()); //Output: [2, 4, 6, 8]
          console.log(numArray.odd()); //Output: [1, 3, 5, 7]
